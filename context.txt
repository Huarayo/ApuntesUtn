🧠 Las 4 operaciones mágicas
1️⃣ RENOMBRAR archivos (updateFileNames)
¿Qué hace?
Cuando cambiás el nombre de un archivo en Drive, el script detecta ese cambio y lo aplica en tu JSON.

¿Cómo sabe?
Cada archivo tiene un ID único (como un DNI). El ID no cambia aunque cambies el nombre. El script busca el ID en Drive y si el nombre es diferente, lo actualiza.

Analogía:
Imaginate que tenés una lista de alumnos con sus nombres. Uno de ellos se cambia el nombre. Vos no lo perdés porque lo identificás por su DNI (ID). Cuando ves que el DNI coincide pero el nombre cambió, actualizás la lista.

2️⃣ RENOMBRAR carpetas (updateFolderNames)
¿Qué hace?
Cuando cambiás el nombre de una carpeta en Drive, el script detecta ese cambio y lo aplica en tu JSON.

¿Cómo sabe?
Las carpetas también tienen un ID único. El script busca la carpeta por su ID en Drive y si el nombre cambió, lo actualiza.

Analogía:
Es lo mismo que con los archivos, pero con carpetas. La carpeta tiene un ID que no cambia, solo cambia el nombre. El script actualiza el nombre manteniendo el ID.

3️⃣ ELIMINAR archivos (removeDeletedNodes)
¿Qué hace?
Cuando eliminás un archivo de Drive, el script lo detecta y lo ELIMINA de tu JSON.

¿Cómo sabe?
El script compara los IDs de Drive con los IDs de tu JSON. Si un ID está en tu JSON pero NO en Drive → el archivo fue eliminado → lo borra.

Analogía:
Tenés una lista de invitados a una fiesta. Cuando alguien confirma que NO va, lo tachás de la lista. El script hace lo mismo: si el archivo ya no está en Drive, lo tacha de tu JSON.

4️⃣ AGREGAR archivos nuevos (findAllFilesWithPath + findFolderByPath)
¿Qué hace?
Cuando subís un archivo nuevo a Drive, el script lo detecta y lo AGREGA a tu JSON en la carpeta correcta.

¿Cómo sabe dónde ponerlo?
El script busca la RUTA completa del archivo en Drive (ej: BÁSICAS/ANÁLISIS MATEMÁTICO I/Exámenes) y busca esa misma ruta en tu JSON. Si la carpeta existe, agrega el archivo ahí.

Analogía:
Llega un paquete nuevo a tu casa. El remitente dice: "Este paquete va en la habitación de arriba, en el segundo cajón". Vos lo ponés exactamente donde dice. El script hace lo mismo: si la carpeta existe, pone el archivo ahí.

function updateFileNames(nodes, driveTree) {
  // 1. Crear un mapa: ID → NUEVO NOMBRE (desde Drive)
  const fileMap = new Map();
  function collectFileNames(driveNodes) {
    for (const node of driveNodes) {
      if (node.type !== "folder" && node.id) {
        fileMap.set(node.id, node.name);  // ← Guarda ID y nuevo nombre
      }
      if (node.children) {
        collectFileNames(node.children);
      }
    }
  }
  collectFileNames(driveTree);

  // 2. Recorrer tu árbol y actualizar nombres
  function updateNames(nodes) {
    for (const node of nodes) {
      if (node.type !== "folder" && node.id && fileMap.has(node.id)) {
        const newName = fileMap.get(node.id);  // ← Obtiene el nuevo nombre
        if (node.name !== newName) {
          console.log(`📝 Archivo renombrado: ${node.name} → ${newName}`);
          node.name = newName;  // ← 🔥 AQUÍ SE CAMBIA EL NOMBRE
        }
      }
      if (node.children) {
        updateNames(node.children);
      }
    }
  }
  updateNames(nodes);
  return nodes;  // ← Devuelve el árbol con nombres actualizados
}

function removeDeletedNodes(nodes, deletedCount = { value: 0 }) {
  const result = [];  // ← Nuevo árbol (sin los eliminados)
  for (const node of nodes) {
    // Si es archivo de Drive y su ID NO está en Drive → ELIMINAR
    if (node.type !== "folder" && node.id && !driveIds.has(node.id)) {
      console.log(`🗑️ Eliminado: ${node.name} (ya no está en Drive)`);
      deletedCount.value++;  // ← Cuenta cuántos eliminó
      continue;  // ← 🔥 SALTA este nodo (NO lo agrega al resultado)
    }
    
    // Si es carpeta, procesar hijos recursivamente
    if (node.type === "folder") {
      if (node.children) {
        const filteredChildren = removeDeletedNodes(node.children, deletedCount);
        result.push({ ...node, children: filteredChildren });
      } else {
        result.push(node);
      }
    } else {
      result.push(node);  // ← Archivo que NO se elimina
    }
  }
  return result;  // ← Devuelve el árbol SIN los eliminados
}

function findAllFilesWithPath(nodes, currentPath = []) {
  let results = [];
  for (const node of nodes) {
    if (node.type === "folder") {
      // Si es carpeta, guardar la ruta y seguir adentro
      if (node.children) {
        const childResults = findAllFilesWithPath(node.children, [...currentPath, node.name]);
        results = results.concat(childResults);
      }
    } else {
      // Si es archivo, guardarlo con su ruta completa
      results.push({
        ...node,
        drivePath: currentPath,  // ← RUTA COMPLETA en Drive
        folderName: currentPath.length > 0 ? currentPath[currentPath.length - 1] : "Raíz"
      });
    }
  }
  return results;
}

function findFolderByPath(nodes, pathParts) {
  let current = nodes;  // ← Empieza desde la raíz
  for (const part of pathParts) {
    const normalizedPart = normalizeName(part);
    // Buscar en el nivel actual
    const found = current.find(node => 
      node.type === "folder" && normalizeName(node.name) === normalizedPart
    );
    if (!found) return null;  // ← Si no existe la ruta, devolver null
    current = found.children || [];  // ← Avanza al siguiente nivel
  }
  return current;  // ← Devuelve la carpeta en la posición exacta
}

import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  // 🔥 Obtener las credenciales del header
  const authHeader = request.headers.get('authorization');
  
  if (!authHeader) {
    // ❌ Sin credenciales → pedir login
    return new NextResponse('Authentication required', {
      status: 401,
      headers: { 'WWW-Authenticate': 'Basic realm="Admin"' },
    });
  }

  // 🔥 Decodificar credenciales
  const base64 = authHeader.split(' ')[1];
  const [user, pass] = Buffer.from(base64, 'base64').toString().split(':');
  
  // 🔥 Verificar usuario y contraseña
  const validUser = process.env.ADMIN_USER || 'admin';
  const validPass = process.env.ADMIN_PASS || 'admin123';
  
  if (user === validUser && pass === validPass) {
    return NextResponse.next(); // ✅ Permitir acceso
  }
  
  // ❌ Credenciales incorrectas
  return new NextResponse('Invalid credentials', { status: 401 });
}

export const config = {
  matcher: '/admin/:path*', // ← Protege SOLO /admin

const ADMIN_USER = process.env.ADMIN_USER || "admin";

export function middleware(request: NextRequest) {
  // 1️⃣ SOLO proteger la ruta /admin
  if (!request.nextUrl.pathname.startsWith('/admin')) {
    return NextResponse.next(); // ← Dejar pasar todo lo que no sea /admin
  }

      status: 401,

        'WWW-Authenticate': 'Basic realm="Panel de Administración"',

  const base64 = authHeader.split(' ')[1];

  // 5️⃣ Verificar usuario y contraseña
  if (user === ADMIN_USER && pass === ADMIN_PASS) {
    return NextResponse.next(); // ✅ Permiso concedido
  }

  // 6️⃣ Credenciales incorrectas
  return new NextResponse('🔒 Usuario o contraseña incorrectos', {
    status: 401,
    headers: {
      'WWW-Authenticate': 'Basic realm="Panel de Administración"',
    },
  });

import { put } from '@vercel/blob';
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

async function uploadToBlob() {
  try {
    // 1. Leer el JSON generado
    const jsonPath = path.join(__dirname, 'data', 'drive-tree.json');
    const jsonContent = fs.readFileSync(jsonPath, 'utf-8');
    const tree = JSON.parse(jsonContent);

    // 2. Subir a Vercel Blob
    const { url } = await put('drive-tree.json', JSON.stringify(tree), {
      access: 'public',
      addRandomSuffix: false,
      contentType: 'application/json',
    });

    console.log(`✅ JSON subido a: ${url}`);
    return url;
  } catch (error) {
    console.error('❌ Error subiendo a Blob:', error);
    process.exit(1);
  }
}

import fs from 'fs';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

async function uploadToBlob() {

export type TreeNode = {
  id?: string;

// ✅ Leer desde Vercel Blob
const BLOB_URL = process.env.NEXT_PUBLIC_BLOB_URL || "https://blob.vercel-storage.com";
const TREE_PATH = "drive-tree.json";

let cachedTree: TreeNode[] | null = null;
let cachedPromise: Promise<TreeNode[]> | null = null;

async function fetchTreeOnce(): Promise<TreeNode[]> {
  if (cachedTree) return cachedTree;

  if (!cachedPromise) {
    const url = `${BLOB_URL}/${TREE_PATH}?t=${Date.now()}`;
    
    cachedPromise = fetch(url, { cache: "no-store" })
      .then((r) => {
        if (!r.ok) throw new Error("Error al cargar el árbol");
        return r.json();
      })
      .then((data: TreeNode[]) => {
        cachedTree = data;
        return data;
      })
      .catch((err) => {
        cachedPromise = null;
        throw err;
      });
  }

  return cachedPromise;
}

export function useTree() {
  const [tree, setTree] = useState<TreeNode[] | null>(cachedTree);

  useEffect(() => {
    if (!tree) {
      fetchTreeOnce().then(setTree).catch(console.error);
    }
  }, [tree]);

  return tree;
}

import { NextResponse } from "next/server";
import { put } from "@vercel/blob";
import { exec } from "child_process";
import { promisify } from "util";
import fs from "fs";
import path from "path";

const execPromise = promisify(exec);

export async function POST(req: Request) {
  try {
    // 🔐 Verificar autenticación
    const authHeader = req.headers.get("authorization");
    const secret = process.env.WEBHOOK_SECRET;
    
    if (!secret || authHeader !== `Bearer ${secret}`) {
      return NextResponse.json({ error: "No autorizado" }, { status: 401 });
    }

    console.log("📢 Webhook recibido - Actualizando desde Drive");

    // 📡 Ejecutar scripts
    console.log("📡 Ejecutando scripts...");
    await execPromise("node scripts/index.js");
    await execPromise("node scripts/merge-drive.js");

    // 📖 Leer el JSON generado
    const jsonPath = path.join(process.cwd(), "scripts", "data", "drive-tree.json");
    const jsonContent = fs.readFileSync(jsonPath, "utf-8");
    const tree = JSON.parse(jsonContent);

    // ☁️ Subir a Vercel Blob
    console.log(`☁️ Subiendo ${tree.length} nodos a Vercel Blob...`);
    const { url } = await put("drive-tree.json", JSON.stringify(tree), {
      access: "public",
      addRandomSuffix: false,
      contentType: "application/json",
    });

    console.log(`✅ JSON actualizado en: ${url}`);

    return NextResponse.json({ 
      success: true, 
      url,
      nodos: tree.length,
      timestamp: new Date().toISOString()
    });

  } catch (error: unknown) {
    console.error("❌ Error:", error);
    if (error instanceof Error) {
      return NextResponse.json({ error: error.message }, { status: 500 });
    }
    return NextResponse.json({ error: "Error desconocido" }, { status: 500 });
  }
}

      .catch((err: Error) => {
        cachedPromise = null;
        throw err;
      });
  }

  return cachedPromise;
}

export function useTree() {
  const [tree, setTree] = useState<TreeNode[] | null>(cachedTree);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    if (!tree) {
      fetchTreeOnce()
        .then(setTree)
        .catch(setError);
    }
  }, [tree]);

  if (error) {
    console.error("Error cargando árbol:", error);
    return [];
  }

  return tree;
}

  onRename,

  onAddFolder: (parentId?: string) => void;

      {nodes.map((node) => (
        <div
          key={node.id}
          style={{
            marginTop: 8,
            padding: "8px 12px",

            borderRadius: 4,
            border: "1px solid #eee",

                <button onClick={() => onAddFile(node.id)} style={{...}}>📄+</button>
              </>
            )}
            
            {/* 🔧 Botones para todos */}
            <button onClick={() => onRename(node.id!)} style={{...}}>✏️</button>
            <button onClick={() => onDelete(node.id!)} style={{...}}>🗑️</button>
          </div>

              nodes={node.children}
              onAddFolder={onAddFolder}
              onAddFile={onAddFile}
              onDelete={onDelete}
              onRename={onRename}
              level={level + 1}
            />
          )}
        </div>
      ))}

import { put } from '@vercel/blob';
import fs from 'fs';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));

async function uploadToBlob() {
  try {
    const jsonPath = path.join(__dirname, 'data', 'drive-tree.json');
    
    if (!fs.existsSync(jsonPath)) {
      console.error('❌ No existe scripts/data/drive-tree.json');
      console.error('Ejecutá primero: node scripts/index.js && node scripts/merge-drive.js');
      process.exit(1);
    }
    
    const jsonContent = fs.readFileSync(jsonPath, 'utf-8');
    const tree = JSON.parse(jsonContent);

    console.log(`☁️ Subiendo ${tree.length} nodos a Vercel Blob...`);
    
    const { url } = await put('drive-tree.json', JSON.stringify(tree), {
      access: 'public',
      addRandomSuffix: false,
      contentType: 'application/json',
    });

    console.log(`✅ JSON subido a: ${url}`);
    console.log(`📊 Nodos: ${tree.length}`);
    return url;

    console.error('❌ Error subiendo a Blob:', error);
    process.exit(1);
  }
}

📁 1. app/middleware.ts (protege /admin)
typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

const ADMIN_USER = process.env.ADMIN_USER || "admin";
const ADMIN_PASS = process.env.ADMIN_PASS || "admin123";

export function middleware(request: NextRequest) {
  if (!request.nextUrl.pathname.startsWith('/admin')) {
    return NextResponse.next();
  }

  const authHeader = request.headers.get('authorization');
  
  if (!authHeader) {
    return new NextResponse('🔒 Acceso restringido', {
      status: 401,
      headers: { 'WWW-Authenticate': 'Basic realm="Panel de Administración"' },
    });
  }

  const base64 = authHeader.split(' ')[1];
  const decoded = Buffer.from(base64, 'base64').toString();
  const [user, pass] = decoded.split(':');

  if (user === ADMIN_USER && pass === ADMIN_PASS) {
    return NextResponse.next();
  }

  return new NextResponse('🔒 Usuario o contraseña incorrectos', {
    status: 401,
    headers: { 'WWW-Authenticate': 'Basic realm="Panel de Administración"' },
  });
}

    for (const node of nodes) {
      if (node.type === "folder") {

        result.push({ ...node, children: filteredChildren });
      } else if (node.source !== "drive") {
        result.push(node);
      }
